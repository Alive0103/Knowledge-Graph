查看json
```bash
# 查看前3行的JSON结构
head -n 3 top_k.jsonl | python -m json.tool

head -n 1 top_k.jsonl | python3 -c "
import json, sys
data = json.loads(sys.stdin.read())

def print_structure(obj, indent=0, max_items=3):
    spaces = '  ' * indent
    if isinstance(obj, dict):
        for i, (key, value) in enumerate(obj.items()):
            if i >= max_items and indent > 0:
                print(f'{spaces}... (还有{len(obj)-max_items}个字段)')
                break
            if isinstance(value, (dict, list)) and indent < 2:  # 限制嵌套深度
                print(f'{spaces}{key}: {type(value).__name__}')
                print_structure(value, indent+1, max_items)
            else:
                value_preview = str(value)[:30] + '...' if len(str(value)) > 30 else str(value)
                print(f'{spaces}{key}: {type(value).__name__} = {value_preview}')
    elif isinstance(obj, list):
        print(f'{spaces}list[{len(obj)}]')
        if obj and indent < 2:
            print_structure(obj[0], indent+1, max_items)

print('=== JSON层级结构 ===')
print_structure(data)
"
```
